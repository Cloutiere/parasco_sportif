    <script>
        // --- LOGIQUE DE CALCUL (JAVASCRIPT) ---

        // Références aux éléments du formulaire et des résultats
        const form = document.getElementById('budgetForm');
        const budgetChartCanvas = document.getElementById('budgetChart').getContext('2d');
        let budgetChart;

        // Fonction pour formater un nombre en devise
        function formatCurrency(value) {
            return value.toFixed(2).replace('.', ',') + ' $';
        }
        
        function calculateBudget() {
            // 1. Récupérer toutes les valeurs des champs (en les convertissant en nombres)
            const headCoachRate = parseFloat(document.getElementById('headCoachRate').value) || 0;
            const assistantCoachRate = parseFloat(document.getElementById('assistantCoachRate').value) || 0;
            
            const seasonWeeks = parseFloat(document.getElementById('seasonWeeks').value) || 0;
            const practicesPerWeek = parseFloat(document.getElementById('practicesPerWeek').value) || 0;
            const practiceDuration = parseFloat(document.getElementById('practiceDuration').value) || 0;
            const numGames = parseFloat(document.getElementById('numGames').value) || 0;
            const gameDuration = parseFloat(document.getElementById('gameDuration').value) || 0;

            const playoffWeeks = parseFloat(document.getElementById('playoffWeeks').value) || 0;
            const playoffFinalDays = parseFloat(document.getElementById('playoffFinalDays').value) || 0;
            const playoffFinalsDuration = parseFloat(document.getElementById('playoffFinalsDuration').value) || 0;

            const tournamentBonus = parseFloat(document.getElementById('tournamentBonus').value) || 0;
            const federationFee = parseFloat(document.getElementById('federationFee').value) || 0;

            // 2. Calculer les coûts
            const totalCoachRatePerHour = headCoachRate + assistantCoachRate;

            // Coûts de la saison régulière
            const totalPracticeHoursSeason = seasonWeeks * practicesPerWeek * practiceDuration;
            const costSeasonPractices = totalPracticeHoursSeason * totalCoachRatePerHour;
            
            const totalGameHoursSeason = numGames * gameDuration;
            const costSeasonGames = totalGameHoursSeason * totalCoachRatePerHour;

            // Coûts des séries (playoffs)
            // Les entraînements continuent au même rythme pendant les séries
            const totalPracticeHoursPlayoffs = playoffWeeks * practicesPerWeek * practiceDuration;
            const costPlayoffPractices = totalPracticeHoursPlayoffs * totalCoachRatePerHour;
            
            const totalFinalsHoursPlayoffs = playoffFinalDays * playoffFinalsDuration;
            const costPlayoffFinals = totalFinalsHoursPlayoffs * totalCoachRatePerHour;

            // Calculs des totaux
            const totalCoachingCost = costSeasonPractices + costSeasonGames + costPlayoffPractices + costPlayoffFinals;
            const grandTotal = totalCoachingCost + tournamentBonus + federationFee;

            // 3. Afficher les résultats dans le tableau
            document.getElementById('resSeasonPractices').innerText = formatCurrency(costSeasonPractices);
            document.getElementById('resSeasonGames').innerText = formatCurrency(costSeasonGames);
            document.getElementById('resPlayoffPractices').innerText = formatCurrency(costPlayoffPractices);
            document.getElementById('resPlayoffFinals').innerText = formatCurrency(costPlayoffFinals);
            document.getElementById('resTotalCoaching').innerText = formatCurrency(totalCoachingCost);
            document.getElementById('resTournament').innerText = formatCurrency(tournamentBonus);
            document.getElementById('resFederation').innerText = formatCurrency(federationFee);
            document.getElementById('resGrandTotal').innerText = formatCurrency(grandTotal);
            
            // 4. Mettre à jour le graphique
            updateChart([costSeasonPractices, costSeasonGames, costPlayoffPractices, costPlayoffFinals, tournamentBonus, federationFee]);
        }

        function updateChart(dataValues) {
            const chartData = {
                labels: [
                    'Salaires - Entraînements Saison',
                    'Salaires - Matchs Saison',
                    'Salaires - Entraînements Séries',
                    'Salaires - Finales Séries',
                    'Bonus Tournoi',
                    'Frais Fédération'
                ],
                datasets: [{
                    label: 'Répartition des coûts',
                    data: dataValues,
                    backgroundColor: [
                        '#007bff',
                        '#17a2b8',
                        '#28a745',
                        '#6f42c1',
                        '#fd7e14',
                        '#dc3545'
                    ],
                    borderColor: '#fff',
                    borderWidth: 2
                }]
            };

            if (budgetChart) {
                // Si le graphique existe déjà, on met juste ses données à jour
                budgetChart.data = chartData;
                budgetChart.update();
            } else {
                // Sinon, on le crée
                budgetChart = new Chart(budgetChartCanvas, {
                    type: 'pie', // Type de graphique : 'pie' (camembert), 'bar' (barres), etc.
                    data: chartData,
                    options: {
                        responsive: true,
                        plugins: {
                            legend: {
                                position: 'top',
                            },
                            title: {
                                display: false,
                                text: 'Répartition du Budget'
                            }
                        }
                    }
                });
            }
        }
        
        // --- Événements ---
        // Lancer le calcul au chargement de la page
        document.addEventListener('DOMContentLoaded', calculateBudget);
        // Lancer le calcul à chaque modification d'un champ du formulaire
        form.addEventListener('input', calculateBudget);

    </script>
